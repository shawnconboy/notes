File I/O in C++

What is fstream?

fstream is part of the C++ Standard Library and allows you to read from and write to files. It's found in the <fstream> header and gives you control over file input/output (I/O).

Writing to a File

This example writes favorite foods to a file called foods.txt:


#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ofstream outFile("foods.txt");

    if (outFile.is_open()) {
        outFile << "Pizza\n";
        outFile << "Sushi\n";
        outFile << "Ice Cream\n";
        outFile.close();
    } else {
        cout << "Failed to open file." << endl;
    }

    return 0;
}

Reading from a File

This reads and prints the contents of foods.txt:


#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream inFile("foods.txt");
    string food;

    if (inFile.is_open()) {
        while (getline(inFile, food)) {
            cout << "I love: " << food << endl;
        }
        inFile.close();
    } else {
        cout << "Unable to open file." << endl;
    }

    return 0;
}
